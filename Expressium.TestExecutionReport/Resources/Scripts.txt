    <script>
        function filterScenarios() {
            var input = document.getElementById("scenario-filter");
            var filterTokens = input.value.toUpperCase().split(/\s+/);
            var table = document.getElementById("scenario-list");
            var tr = table.getElementsByTagName("tr");

            for (var i = 0; i < tr.length; i++) {
                var role = tr[i].getAttribute("role");
                var tags = tr[i].getAttribute("tags");
                var td = tr[i].getElementsByTagName("td");
                var rowText = "";
                var found = false;

                if (role == "feature")
				    continue;

                for (var j = 0; j < td.length; j++) {
                    rowText += td[j].textContent.toUpperCase() + " ";
                }
                
                var tagTokens = tags.toUpperCase().split(/\s+/);
                for (var t = 0; t < tagTokens.length; t++) {
                    rowText += tagTokens[t].toUpperCase() + " ";
                }

                found = filterTokens.every(token => rowText.includes(token));

                tr[i].style.display = found ? "" : "none";
            }
        }
    </script>
    <script>
        function sortTableByColumn(n) {
            let table;
            table = document.getElementById("scenariolist");
            var rows, i, x, y, count = 0;
            var switching = true;

            var direction = "ascending";

            while (switching) {
                switching = false;
                var rows = table.rows;

                for (i = 1; i < (rows.length - 1); i++) {
                    var Switch = false;

                    x = rows[i].getElementsByTagName("TD")[n];
                    y = rows[i + 1].getElementsByTagName("TD")[n];

                    if (direction == "ascending") {

                        if (x.innerHTML.toLowerCase() >
                            y.innerHTML.toLowerCase()) {

                            Switch = true;
                            break;
                        }
                    } else if (direction == "descending") {

                        if (x.innerHTML.toLowerCase() <
                            y.innerHTML.toLowerCase()) {

                            Switch = true;
                            break;
                        }
                    }
                }
                if (Switch) {
                    rows[i].parentNode.insertBefore(rows[i + 1], rows[i]);
                    switching = true;

                    count++;
                } else {
                    if (count == 0 && direction == "ascending") {
                        direction = "descending";
                        switching = true;
                    }
                }
            }
        }
    </script>
    <script>
        function showStatusChart() {
            let button = document.getElementById('status-chart-button');
            let div = document.getElementById('status-chart');
            if (div.style.display === "none") {
                div.style.display = "";
                button.title = "Close Status Chart";
                button.innerHTML = "<b>&#9866;</b>";                        
            } else {
                div.style.display = "none";
                button.title = "Open Status Chart";
                button.innerHTML = "<b>&#10011;</b>";    
            }           
        }
    </script>
